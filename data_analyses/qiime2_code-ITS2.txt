# Note: prior to assigning taxonomy a classifier needs to be downloaded and put in classifier folder
# ITS2 classifier: UNITE version 8 dynamic classifier - https://unite.ut.ee/repository.php (or see Mendeley Data)

#----- QIIME 2 environment activation in Conda
conda activate qiime2-2021.8

#----- Change to working directory and set up folders

cd data_analyses
mkdir -p input_data/ITS2_endo_2021
# copy fastq.gz files from source (ENA) to input_data/ITS2_endo_2021
# the sequences were pre-processed by the sequencing company and most non-biological sequences were removed 

# create directory for qiime2 output
mkdir -p qiime2_output/ITS2_endo_2021-outputs

#----- Import ITS2 cloacal and tank water data in Cassava 1.8 format

qiime tools import \
  --type 'SampleData[PairedEndSequencesWithQuality]' \
  --input-path input_data/ITS2_endo_2021 \
  --input-format CasavaOneEightSingleLanePerSampleDirFmt \
  --output-path qiime2_output/ITS2_endo_2021-outputs/demux-paired-end_ITS.qza

qiime demux summarize \
  --i-data qiime2_output/ITS2_endo_2021-outputs/demux-paired-end_ITS.qza \
  --o-visualization qiime2_output/ITS2_endo_2021-outputs/demux-paired-end_ITS.qzv

#----- Remove primers in sequences using cutadapt prior to DADA2, avoiding readthrough due to sequence length variability
#for ITS2
#  --p-adapter-f GCATATCAATAAGCGGAGGA\ reverse primer reverse complement
#  --p-front-f GCATCGATGAAGAACGCAGC\ forward primer ITS3
#  --p-adapter-r GCTGCGTTCTTCATCGATGC\ forward primer reverse complement
#  --p-front-r TCCTCCGCTTATTGATATGC\ reverse primer ITS4

qiime cutadapt trim-paired \
  --i-demultiplexed-sequences qiime2_output/ITS2_endo_2021-outputs/demux-paired-end_ITS.qza \
  --p-adapter-f GCATATCAATAAGCGGAGGA \
  --p-front-f GCATCGATGAAGAACGCAGC \
  --p-adapter-r GCTGCGTTCTTCATCGATGC \
  --p-front-r TCCTCCGCTTATTGATATGC \
  --o-trimmed-sequences qiime2_output/ITS2_endo_2021-outputs/demux-trimmed_ITS.qza \
  --verbose
  
qiime demux summarize \
  --i-data qiime2_output/ITS2_endo_2021-outputs/demux-trimmed_ITS.qza \
  --o-visualization qiime2_output/ITS2_endo_2021-outputs/demux-trimmed_ITS.qzv
  
#----- Denoising sequences with DADA2

qiime dada2 denoise-paired \
  --i-demultiplexed-seqs qiime2_output/ITS2_endo_2021-outputs/demux-trimmed_ITS.qza \
  --p-trim-left-f 0 \
  --p-trim-left-r 0 \
  --p-trunc-len-f 257 \
  --p-trunc-len-r 178 \
  --o-table qiime2_output/ITS2_endo_2021-outputs/table-trimmed.qza \
  --o-representative-sequences qiime2_output/ITS2_endo_2021-outputs/rep-seqs-trimmed.qza \
  --o-denoising-stats qiime2_output/ITS2_endo_2021-outputs/denoising-stats-trimmed.qza \
  --verbose

qiime metadata tabulate \
  --m-input-file qiime2_output/ITS2_endo_2021-outputs/denoising-stats-trimmed.qza \
  --o-visualization qiime2_output/ITS2_endo_2021-outputs/denoising-stats-trimmed.qzv

qiime feature-table summarize \
  --i-table qiime2_output/ITS2_endo_2021-outputs/table-trimmed.qza \
  --o-visualization qiime2_output/ITS2_endo_2021-outputs/table-trimmed.qzv \
  --m-sample-metadata-file qiime2_output/ITS2_endo_2021-outputs/metadata-ITS-new.tsv
  
qiime feature-table tabulate-seqs \
  --i-data qiime2_output/ITS2_endo_2021-outputs/rep-seqs-trimmed.qza \
  --o-visualization qiime2_output/ITS2_endo_2021-outputs/rep-seqs-trimmed.qzv
  
#----- Align sequences using mafft and fasttree

qiime phylogeny align-to-tree-mafft-fasttree \
  --i-sequences qiime2_output/ITS2_endo_2021-outputs/rep-seqs-trimmed.qza \
  --o-alignment qiime2_output/ITS2_endo_2021-outputs/aligned-rep-seqs.qza \
  --o-masked-alignment qiime2_output/ITS2_endo_2021-outputs/masked-aligned-rep-seqs.qza \
  --o-tree qiime2_output/ITS2_endo_2021-outputs/unrooted-tree.qza \
  --o-rooted-tree qiime2_output/ITS2_endo_2021-outputs/rooted-tree.qza

#----- Assigning taxonomy via UNITE database v.8 dynamic classifier (premade by Qiime 2 team)

qiime feature-classifier classify-sklearn \
  --i-classifier classifiers/unite_classifier_ver8_dynamic.qza \
  --i-reads qiime2_output/rep-seqs-trimmed.qza \
  --o-classification qiime2_output/taxonomy.qza

qiime metadata tabulate \
  --m-input-file qiime2_output/taxonomy.qza \
  --o-visualization qiime2_output/taxonomy.qzv

qiime taxa barplot \
  --i-table qiime2_output/table-trimmed.qza \
  --i-taxonomy qiime2_output/taxonomy.qza \
  --m-metadata-file qiime2_output/metadata-ITS-new.tsv \
  --o-visualization qiime2_output/taxa-bar-plots.qzv

#----- Alpha rarefaction to determine approapriate depth for downstream analyses

qiime diversity alpha-rarefaction \
  --i-table qiime2_output/ITS2_endo_2021-outputs/table-trimmed.qza \
  --i-phylogeny qiime2_output/ITS2_endo_2021-outputs/rooted-tree.qza \
  --p-max-depth 69203 \
  --m-metadata-file qiime2_output/ITS2_endo_2021-outputs/metadata-ITS-new.tsv \
  --o-visualization qiime2_output/ITS2_endo_2021-outputs/alpha-rarefaction.qzv

  --p-where "[SampleSite]='CLOACA'" \

  --p-where "[SampleSite]='WATER'" \
